CREATE TABLE gss.TB_ORGANIZATION(
	ORG_ID VARCHAR(40) NOT NULL PRIMARY KEY,
 	ORG_NAME VARCHAR(40) DEFAULT NULL,
 	ORG_ADDRESS VARCHAR(40) DEFAULT NULL,
 	ORG_NUMBER VARCHAR(40) DEFAULT NULL,
 	ORG_PAX_NUMBER VARCHAR(40) DEFAULT null,
        ORG_EMAIL VARCHAR(40) DEFAULT null,
        ORG_ADDRESS_DETAIL VARCHAR(40) DEFAULT null,
        ORG_ZONE_CODE VARCHAR(40) DEFAULT null,
        ORG_ETC VARCHAR(40) DEFAULT null,
 	CREATE_DATE TIMESTAMP DEFAULT NULL,
	CREATE_USER_ID VARCHAR(40) DEFAULT NULL,
	UPDATE_DATE TIMESTAMP DEFAULT NULL,
	UPDATE_USER_ID VARCHAR(40) DEFAULT NULL,
	USE_YN VARCHAR(2)
);

CREATE TABLE gss.TB_ORDER(
	ORDER_ID VARCHAR(40) NOT NULL PRIMARY KEY,
	ORDER_ORG_ID VARCHAR(40) DEFAULT NULL,
	ORDER_ORDERING_DATE DATE DEFAULT NULL,
	ORDER_DEAD_LINE_DATE DATE DEFAULT NULL,
	USE_YN VARCHAR(2) DEFAULT NULL,
	CREATE_DATE TIMESTAMP DEFAULT NULL,
	CREATE_USER_ID VARCHAR(40) DEFAULT NULL,
	UPDATE_DATE TIMESTAMP DEFAULT NULL,
	UPDATE_ID VARCHAR(40) DEFAULT NULL
);

CREATE TABLE gss.TB_PRODUCT(
	PRODUCT_ID VARCHAR(40) NOT null PRIMARY KEY,
	ORDER_ID VARCHAR(40) DEFAULT NULL,
	PRODUCT_STYLE_NO VARCHAR(40) DEFAULT NULL,
	PRODUCT_ITEM VARCHAR(40) DEFAULT NULL,
	PRODUCT_SIZE VARCHAR(40) DEFAULT NULL,
	PRODUCT_COLOR VARCHAR(40) DEFAULT NULL,
	PRODUCT_QTY VARCHAR(40) DEFAULT NULL,
	PRODUCT_PRICE VARCHAR(40) DEFAULT NULL,
	PRODUCT_ETC VARCHAR(40) DEFAULT NULL,
	USE_YN VARCHAR(2) DEFAULT NULL,
	CREATE_DATE TIMESTAMP DEFAULT NULL,
	CREATE_USER_ID VARCHAR(40) DEFAULT NULL,
	UPDATE_DATE TIMESTAMP DEFAULT NULL,
	UPDATE_ID VARCHAR(40) DEFAULT NULL
);

CREATE TABLE gss.TB_AUTHORITY (
	AUTHORITY_ID VARCHAR(100) PRIMARY KEY ,
	AUTHORITY_CODE VARCHAR(30) ,
	AUTHORITY_NAME VARCHAR(100) ,
	AUTHORITY_DESC VARCHAR(200) ,
	USE_YN VARCHAR(2) DEFAULT NULL ,
	CREATE_DATE TIMESTAMP DEFAULT NULL ,
	CREATE_USER_ID VARCHAR(40) DEFAULT NULL ,
	UPDATE_DATE TIMESTAMP DEFAULT NULL ,
	UPDATE_ID VARCHAR(40) DEFAULT NULL
);


CREATE TABLE gss.TB_GSS_USER (
	USER_ID VARCHAR(100) PRIMARY KEY,
	USER_PASSWORD VARCHAR(100),
	USER_NAME VARCHAR(20),
	USER_PHONE_NUMBER VARCHAR(30),
	USER_ADDRESS VARCHAR(100),
	USER_EMAIL VARCHAR (100),
	USE_YN VARCHAR(2) DEFAULT NULL ,
	CREATE_DATE TIMESTAMP DEFAULT NULL ,
	CREATE_USER_ID VARCHAR(40) DEFAULT NULL ,
	UPDATE_DATE TIMESTAMP DEFAULT NULL ,
	UPDATE_ID VARCHAR(40) DEFAULT NULL
);

CREATE TABLE gss.GRANTED_AUTHORITY (
	USER_ID VARCHAR(100) ,
	AUTHORITY_ID VARCHAR(100) ,
	USE_YN VARCHAR(2) DEFAULT NULL ,
	CREATE_DATE TIMESTAMP DEFAULT NULL ,
	CREATE_USER_ID VARCHAR(40) DEFAULT NULL ,
	UPDATE_DATE TIMESTAMP DEFAULT NULL ,
	UPDATE_ID VARCHAR(40) DEFAULT NULL ,
	PRIMARY KEY(USER_ID, AUTHORITY_ID)
);

CREATE TABLE gss.TB_COMMON_GROUP (
	COMMON_GROUP_ID VARCHAR(100) PRIMARY KEY,
	COMMON_GROUP_NAME VARCHAR (100),
	COMMON_GROUP_DESC VARCHAR (100),
	USE_YN VARCHAR(2) DEFAULT NULL ,
	CREATE_DATE TIMESTAMP DEFAULT NULL ,
	CREATE_USER_ID VARCHAR(40) DEFAULT NULL ,
	UPDATE_DATE TIMESTAMP DEFAULT NULL ,
	UPDATE_ID VARCHAR(40) DEFAULT NULL
);

CREATE TABLE gss.TB_COMMON(
	COMMON_ID VARCHAR(100) ,
	COMMON_GROUP_ID VARCHAR(100) ,
	COMMON_NAME VARCHAR (100),
	COMMON_DESC VARCHAR (100) ,
	USE_YN VARCHAR(2) DEFAULT NULL ,
	CREATE_DATE TIMESTAMP DEFAULT NULL ,
	CREATE_USER_ID VARCHAR(40) DEFAULT NULL ,
	UPDATE_DATE TIMESTAMP DEFAULT NULL ,
	UPDATE_ID VARCHAR(40) DEFAULT NULL,
